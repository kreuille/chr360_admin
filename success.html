// ── Init ───────────────────────────────────────────────────── async function init() { const url = new URLSearchParams(window.location.search); const sessionId = url.get('session_id'); const canceled = url.get('canceled'); const activated = url.get('activated'); // client existant qui active son abonnement const token = url.get('token'); // token dashboard du client existant if (canceled) { window.location.href = 'inscription.html?canceled=1'; return; } if (!sessionId) { window.location.href = 'inscription.html'; return; } // ── Cas : client existant qui ajoute sa carte / Réactive ── if (activated === '1' && token) { document.getElementById('progressSection').style.display = 'block'; document.getElementById('progressLabel').textContent = 'Synchronisation avec Stripe...'; document.getElementById('progressBar').style.width = '50%'; // Boucle de vérification (Polling) pour s'assurer que le backend a mis à jour l'état let attempts = 0; const syncInterval = setInterval(async () => { attempts++; try { const resp = await fetchAPI({action: 'syncSubscription', id: token}); if (resp && resp.synced) { clearInterval(syncInterval); document.getElementById('progressBar').style.width = '100%'; document.getElementById('progressLabel').textContent = 'Abonnement activé ✅'; setTimeout(() => { window.location.href = 'dashboard.html?id=' + token; }, 1000); } else if (attempts >= 6) { // Après 12 secondes, on force la redirection clearInterval(syncInterval); window.location.href = 'dashboard.html?id=' + token; } } catch(e) { if (attempts >= 6) { clearInterval(syncInterval); window.location.href = 'dashboard.html?id=' + token; } } }, 2000); // Vérifie toutes les 2 secondes return; } startAnimation(); // ── Cas : nouveau client (1ère installation) ── const params = { action: 'registerTrial', session_id: sessionId, nom: url.get('nom') || '', email: url.get('email') || '', restaurant: url.get('restaurant') || '', tel: url.get('tel') || '', devise: url.get('devise') || 'CHF', plan: url.get('plan') || 'starter', }; try { const resp = await installClient(params); if (resp && resp.timeout) { showFinal(params.email, null); return; } if (resp && resp.success) { // ✅ Utilise l'URL retournée par le serveur pour éviter de la deviner const dashUrl = resp.dashboard_url || null; showFinal(params.email, dashUrl); } else { showFinal(params.email, null); } } catch(e) { showFinal(params.email, null); } } // Fonction Helper pour faire des requêtes JSONP propres dans success.html function fetchAPI(params) { return new Promise((resolve, reject) => { const cb = '_api_' + Date.now(); let done = false; window[cb] = d => { if(!done){done=true;delete window[cb];resolve(d);} }; const qs = Object.entries(params).map(([k,v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`).join('&'); const s = document.createElement('script'); s.src = APPS_SCRIPT_URL + '?' + qs + '&callback=' + cb; s.onerror = () => { if(!done){done=true;delete window[cb];s.remove();reject(new Error('Network error'));} }; document.head.appendChild(s); setTimeout(() => { if(!done){done=true;delete window[cb];s.remove();reject(new Error('Timeout'));} }, 10000); }); }
